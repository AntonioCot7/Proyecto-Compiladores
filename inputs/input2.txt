#include <stdio.h>

typedef unsigned int uint;

typedef struct {
    int x;
    int y;
    int energy;
} Particle;

int main() {
    Particle p;
    int time;
    
    p.x = 0;
    p.y = 0;
    p.energy = 100;
    time = 0;
    
    while (p.energy > 0) {
        if (time > 5) {
            p.energy = 0;
        }
        
        for (int i = 0; i < 3; i++) {
            p.x = p.x + 1;
            p.y = (p.x > p.y) ? (p.y + 2) : (p.y + 1);
            
            if (p.x == p.y) {
                printf("Colision en %d,%d\n", p.x, p.y);
                p.energy = p.energy - 20;
            }
        }
        
        printf("T=%d Pos(%d,%d) E=%d\n", time, p.x, p.y, p.energy);
        time = time + 1;
        p.energy = p.energy - 10;
    }
    
    printf("Simulacion terminada.\n");
    return 0;
}
