#include <stdio.h>
typedef struct {
    int id;
    money balance;
} Account;
typedef int money;
Account withdraw(Account acc, money amount) {
int can_withdraw;
can_withdraw = ((acc.balance >= amount) ? 1 : 0);
if (can_withdraw) {
acc.balance = (acc.balance - amount);
printf("Retiro exitoso. Nuevo saldo: %d\n", acc.balance);
}
 else {
printf("Fondos insuficientes en cuenta %d\n", acc.id);
}
return acc;
}
int main() {
Account alice;
alice.id = 1;
alice.balance = 1000;
printf("Saldo inicial: %d\n", alice.balance);
alice = withdraw(alice, 300);
printf("Saldo tras retiro 1: %d\n", alice.balance);
alice = withdraw(alice, 800);
printf("Saldo final: %d\n", alice.balance);
return 0;
}
Interprete:
Saldo inicial: 1000
Retiro exitoso. Nuevo saldo: 700
Saldo tras retiro 1: 700
Fondos insuficientes en cuenta 1
Saldo final: 700
