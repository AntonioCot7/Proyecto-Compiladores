#include <stdio.h>
int ack(int m, int n) {
if ((m == 0)) {
return (n + 1);
}
if ((n == 0)) {
return ack((m - 1), 1);
}
return ack((m - 1), ack(m, (n - 1)));
}
int main() {
int m;
int n;
m = 0;
while ((m <= 3)) {
n = 0;
while ((n <= 2)) {
printf("Ack(%d, %d) = %d\n", m, n, ack(m, n));
n = (n + 1);
}
m = (m + 1);
}
return 0;
}
